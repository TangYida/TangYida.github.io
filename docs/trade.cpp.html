1 #include<bits/stdc++.h>
 2 #define int long long
 3 const int N = 5e3 + 3;
 4 
 5 int n,w;
 6 int size[N];
 7 int ver[N << 1],head[N],nxt[N << 1],edge[N << 1],tot{1};
 8 int f[N][N];
 9 
10 inline void add(int x,int y,int z)
11 {/*{{{*/
12         ver[++tot] = y,edge[tot] = z,nxt[tot] = head[x],head[x] = tot;
13 }/*}}}*/
14 
15 inline int read()
16 {/*{{{*/
17         int s{0},f{1}; char ch;
18         while(!isdigit(ch = getchar())) if(ch == '-') f = -1;
19         while(isdigit(ch)) s = s * 10 + ch - '0',ch = getchar();
20         return s * f;
21 }/*}}}*/
22 
23 inline void chkMax(int& x,int y) {x = std::max(x,y);}
24 
25 inline void dfs(int x,int fa)
26 {
27         size[x] = 1;
28         for(int i = head[x];i;i = nxt[i]) {
29                 /*size[x]实际是子树大小前缀和，写起来更加简洁*/
30                 if(ver[i] == fa) continue;
31                 dfs(ver[i],x);
32 
33                 #define __f(i,up) for(int (i) = up;(i) >= 0;(i)--)
34                 /*要注意倒序枚举*/
35                 __f(j,size[x]) __f(k,size[ver[i]]) {/*分组背包??*/
36                         if(w - k < 0 || (n - w + k - size[ver[i]]) < 0)
37                                 continue;
38                         chkMax(f[x][j + k],f[x][j] + f[ver[i]][k] +
39                          edge[i] * k * (w - k) +
40                          edge[i] * (size[ver[i]] - k) * (n - w + k - size[ver[i]]));
41                 }
42                 #undef __f
43                 size[x] += size[ver[i]]; /*更新子树大小前缀和*/
44         }
45 }

54         for(int i = 1,u,v,value;i < n;i++) {
55                 u = read(),v = read(), value = read();
