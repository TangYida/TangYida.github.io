inline void init()
{
    for(int i = 0;i <= n + 1;i++) for(int j = 0;j <= n + 1;j++) f[i][j] = g[i][j] = 0;
}

#define __f(i,begin,end) for(int i = begin;i <= end;i++)
#define __df(i,end,begin) for(int i = end;i >= begin;i--)
​
inline void dp(int max)
{
    f[0][0] = 1;
    __f(i,1,n) __f(j,0,i) /*i:枚举第几条鱼*/ /*j:枚举死几个* */ {
        f[i][j] = 1ll * f[i - 1][j] * mod(1 - p[i][max]) % MOD;
        if(j) f[i][j] = (1ll * f[i][j] + f[i - 1][j - 1] * p[i][max] % MOD) % MOD;
    }
    g[n + 1][0] = 1;
    __df(i,n,1) __f(j,0,n - i +1) /*同上*/ {
        g[i][j] = 1ll * g[i + 1][j] * mod(1 - p[i][max]) % MOD;
        if(j) g[i][j] = (1ll * g[i][j] + g[i + 1][j - 1] * p[i][max] % MOD) % MOD;
    }
    /*后缀和*/
    __f(i,1,n) __df(j,n - i,0)
        g[i][j] = (1ll * g[i][j] + g[i][j + 1]) % MOD;
}
